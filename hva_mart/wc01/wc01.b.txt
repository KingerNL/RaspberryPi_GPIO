Logfile (C versie)

Een logfile wordt vaak gebruikt om tijdens het draaien van een programma op 
de hoogte te blijven van de status van de uitvoer. Het is een tekstbestand 
dat steeds groter wordt. Mede door deze eenvoud is het een van de snelste 
manieren om snel tekstuele (soms binaire) data op te slaan.

    printf("\nAfgebroken :(\n\n");
  };

  // Maak 1000 keer
  for (int i = 0; i < 1000; i++) {
    //TODO #2: wijzig de tekst i kwadraad te tonen

// 1.b clogfile

#include "logfile.h"

FILE* logfile_open(char* sFilename) {

  // Open de logfile om aan toe te voegen.
  return fopen(sFilename, "a");
};

bool logfile_close(FILE *pFile) {

  // Sluit het bestand (0 = succes)
  return fclose(pFile) == 0;
};

void logfile_write(FILE *pFile, char* pBuf) {

  char sNewline[] = "\n";

  // Schrijf alle letters.
  while (*pBuf > 0) {
    fputc(*pBuf, pFile);
    pBuf++;
  };

  // Sluit af met \n
  pBuf = sNewline;
  while (*pBuf > 0) {
    fputc(*pBuf, pFile);
    pBuf++;
  };
};
